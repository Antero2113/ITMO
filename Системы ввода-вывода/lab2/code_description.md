
## Файл 1: Makefile (сборка модуля)

```makefile
# Указываем, какой объектный файл нужно собрать
# var1.o будет собран из var1.c
obj-m += var1.o

# Цель "all" - собрать модуль
all:
	# -C /lib/modules/$(shell uname -r)/build
	#   переходит в директорию с ядром текущей системы
	# M=$(PWD) - указывает, где лежит наш исходный код
	# modules - цель сборки модулей ядра
	make -C /lib/modules/$(shell uname -r)/build M=$(PWD) modules

# Цель "clean" - удалить собранные файлы
clean:
	make -C /lib/modules/$(shell uname -r)/build M=$(PWD) clean
```

**Что это значит для новичка:**
- `obj-m` — переменная Makefile, говорит "собрать модуль из указанного файла"
- `$(shell uname -r)` — выполняет команду `uname -r`, которая выводит версию вашего ядра (например, "5.15.0-91-generic")
- Путь `/lib/modules/...` — там лежат файлы ядра и инструменты для сборки модулей

## Файл 2: var1.c (драйвер символьного устройства)

```c
// ============================================================
// БЛОК 1: ПОДКЛЮЧЕНИЕ ЗАГОЛОВОЧНЫХ ФАЙЛОВ
// ============================================================
// Заголовочные файлы ядра Linux - содержат объявления функций,
// констант и структур, которые мы будем использовать

#include <linux/module.h>    // Обязательный для всех модулей ядра
#include <linux/kernel.h>    // Определения типов, макросы (printk, KERN_INFO)
#include <linux/fs.h>        // Файловая система: регистрация символьных устройств
#include <linux/cdev.h>      // Обработка символьных устройств (cdev)
#include <linux/device.h>    // Создание классов и устройств в /sys и /dev
#include <linux/uaccess.h>   // copy_to_user, copy_from_user - безопасный обмен данными
#include <linux/slab.h>      // kmalloc, kfree, krealloc - управление памятью в ядре

// ============================================================
// БЛОК 2: ОПРЕДЕЛЕНИЯ И ГЛОБАЛЬНЫЕ ПЕРЕМЕННЫЕ
// ============================================================
// Это константы, которые будут видны из пространства пользователя

#define DEVICE_NAME "var1"      // Имя устройства (будет в /dev/var1)
#define CLASS_NAME "var1_class" // Имя класса устройства (в /sys/class/var1_class/)

// --- ПЕРЕМЕННЫЕ ДЛЯ РАБОТЫ С УСТРОЙСТВОМ ---
static int major;                    // Старший номер устройства (мажорный)
                                     // Например, 240 - уникально идентифицирует драйвер

static struct class *var1_class;     // Указатель на класс устройства
                                     // Класс группирует похожие устройства

static struct device *var1_device;   // Указатель на само устройство
                                     // С ним работает пользователь через /dev/var1

static struct cdev var1_cdev;        // Структура символьного устройства ядра
                                     // cdev = Character DEVice

// --- ДАННЫЕ, КОТОРЫЕ ХРАНИТ ДРАЙВЕР ---
static int *results;    // Указатель на массив целых чисел
                        // В этом массиве хранятся все записанные в устройство длины данных

static int count;       // Количество элементов в массиве results
                        // (сколько раз было вызвано write)

static int capacity;    // Вместимость массива results (сколько памяти выделено)
                        // Если count > capacity, массив расширяется

// ============================================================
// БЛОК 3: ФУНКЦИЯ OPEN - ВЫЗЫВАЕТСЯ ПРИ ОТКРЫТИИ ФАЙЛА УСТРОЙСТВА
// ============================================================
// Пользователь делает: int fd = open("/dev/var1", O_RDWR);
// Ядро вызывает эту функцию

static int var1_open(struct inode *inode, struct file *file)
{
    // inode - содержит информацию о файле (метаданные)
    // file - файловый дескриптор в ядре
    
    // В данном драйвере при открытии ничего не делаем
    // Просто возвращаем 0, что означает "всё хорошо"
    return 0;
}

// ============================================================
// БЛОК 4: ФУНКЦИЯ RELEASE - ВЫЗЫВАЕТСЯ ПРИ ЗАКРЫТИИ ФАЙЛА
// ============================================================
// Пользователь делает: close(fd);
// Ядро вызывает эту функцию

static int var1_release(struct inode *inode, struct file *file)
{
    // Здесь можно было бы освободить ресурсы, но в данном
    // драйвере всё хранится в глобальных переменных,
    // поэтому тоже ничего не делаем
    return 0;
}

// ============================================================
// БЛОК 5: ФУНКЦИЯ WRITE - ПРИЁМ ДАННЫХ ОТ ПОЛЬЗОВАТЕЛЯ
// ============================================================
// Пользователь пишет в устройство: write(fd, buffer, len);
// Это ЗАПИСЫВАЕТ данные в драйвер

static ssize_t var1_write(struct file *file, 
                          const char __user *buf,   // Указатель на данные пользователя
                          size_t len,               // Длина данных, которые записывают
                          loff_t *off)              // Смещение в файле (не используется)
{
    int *new_results;      // Временный указатель для нового массива
    int new_count;         // Новое количество элементов

    // Увеличиваем счётчик элементов на 1 (мы добавим один новый элемент)
    new_count = count + 1;

    // Если новое количество больше вместимости массива
    if (new_count > capacity) {
        // Рассчитываем новую вместимость:
        // если capacity == 0 (первый раз) -> выделяем 8 элементов
        // иначе увеличиваем в 2 раза (capacity * 2)
        capacity = capacity == 0 ? 8 : capacity * 2;
        
        // krealloc - перевыделяет память: увеличивает массив results
        // до размера capacity * sizeof(int)
        new_results = krealloc(results, capacity * sizeof(int), GFP_KERNEL);
        
        // GFP_KERNEL - флаг означает "обычное выделение памяти"
        // Если память закончилась, krealloc вернёт NULL
        if (!new_results)
            return -ENOMEM;  // ENOMEM = Ошибка "Нет памяти"
            
        // Обновляем указатель на массив
        results = new_results;
    }

    // Кладём в массив (в конец) длину len, которую нам передали при записи
    // Например, если пользователь записал 100 байт, мы сохраняем число 100
    results[count] = len;
    
    // Увеличиваем счётчик
    count++;

    // Возвращаем len - количество байт, которые "принял" драйвер
    return len;
}

// ============================================================
// БЛОК 6: ФУНКЦИЯ READ - ПЕРЕДАЧА ДАННЫХ ПОЛЬЗОВАТЕЛЮ
// ============================================================
// Пользователь читает: read(fd, buffer, max_len);
// Драйвер отдаёт ВСЕ сохранённые длины (все результаты write)

static ssize_t var1_read(struct file *file, 
                         char __user *buf,        // Буфер пользователя (куда положить данные)
                         size_t len,              // Сколько пользователь хочет прочитать
                         loff_t *off)             // Смещение (позиция) в файле
{
    char *kbuf;      // Указатель на буфер в ядре (куда сложим данные)
    int i, pos = 0;  // i - счётчик, pos - текущая позиция в буфере
    int size = 0;    // Общий размер буфера, который понадобится

    // *off - смещение от начала файла
    // Если смещение не 0, значит пользователь уже читал данные
    // Возвращаем 0, что означает "данные кончились"
    if (*off > 0)
        return 0;

    // Вычисляем, сколько памяти нужно для вывода всех чисел
    // Каждое число занимает максимум 20 символов (например "-2147483648")
    // Плюс символ перевода строки \n
    for (i = 0; i < count; i++)
        size += 20;   // 20 байт на каждое число + перевод строки

    // Выделяем память в ядре под буфер
    // +1 на завершающий нулевой символ
    kbuf = kmalloc(size + 1, GFP_KERNEL);
    if (!kbuf)
        return -ENOMEM;  // Нет памяти

    // Формируем строку: каждое число из results[i] -> текст ("123\n")
    // sprintf преобразует число в строку и добавляет символ новой строки
    // pos += ... прибавляет к pos длину записанной строки
    for (i = 0; i < count; i++)
        pos += sprintf(kbuf + pos, "%d\n", results[i]);

    // copy_to_user - КОПИРУЕТ ДАННЫЕ ИЗ ЯДРА В ПОЛЬЗОВАТЕЛЯ
    // Это важно! Пользовательский код не может напрямую читать память ядра
    // Возвращает 0 в случае успеха, иначе количество непереданных байт
    if (copy_to_user(buf, kbuf, pos)) {
        kfree(kbuf);           // Освобождаем память ядра
        return -EFAULT;        // EFAULT = Ошибка адресации (Bad address)
    }

    // Освобождаем память ядра (буфер нам больше не нужен)
    kfree(kbuf);
    
    // Сохраняем позицию в файле (сколько байт "прочитано")
    *off = pos;

    // Возвращаем количество переданных пользователю байт
    return pos;
}

// ============================================================
// БЛОК 7: СТРУКТУРА ФАЙЛОВЫХ ОПЕРАЦИЙ
// ============================================================
// Эта структура связывает системные вызовы (open, read, write, close)
// с конкретными функциями нашего драйвера

static struct file_operations fops = {
    .open    = var1_open,      // Когда пользователь вызывает open
    .release = var1_release,   // Когда пользователь вызывает close
    .read    = var1_read,      // Когда пользователь вызывает read
    .write   = var1_write,     // Когда пользователь вызывает write
};

// ============================================================
// БЛОК 8: ИНИЦИАЛИЗАЦИЯ МОДУЛЯ (загружается при insmod)
// ============================================================
// __init - макрос, который говорит: эта функция нужна только при загрузке
// Её память можно освободить после инициализации

static int __init var1_init(void)
{
    dev_t dev;  // dev = device number (комбинированный major:minor номер)

    // 1. Регистрируем символьное устройство
    //    alloc_chrdev_region - ядро само выделит свободный мажорный номер
    //    &dev - сюда запишется номер устройства
    //    0 - первый минорный номер
    //    1 - количество устройств (одно)
    //    DEVICE_NAME - имя устройства
    alloc_chrdev_region(&dev, 0, 1, DEVICE_NAME);
    
    // 2. Извлекаем мажорный номер (старший) из комбинированного числа dev
    major = MAJOR(dev);  // Например, из 240:0 получаем 240

    // 3. Инициализируем структуру cdev
    cdev_init(&var1_cdev, &fops);  // Привязываем file_operations к устройству
    
    // 4. Добавляем устройство в систему
    cdev_add(&var1_cdev, dev, 1);  // dev - номер устройства, 1 - количество

    // 5. Создаём класс (группировку устройств)
    var1_class = class_create(THIS_MODULE, CLASS_NAME);
    
    // 6. Создаём устройство (появляется в /dev/var1)
    var1_device = device_create(var1_class, NULL, dev, NULL, DEVICE_NAME);

    // 7. Инициализируем хранилище данных
    results = NULL;   // Пока нет массива
    count = 0;        // 0 элементов
    capacity = 0;     // Вместимость 0

    return 0;  // Успешная инициализация
}

// ============================================================
// БЛОК 9: ЗАВЕРШЕНИЕ МОДУЛЯ (выгружается при rmmod)
// ============================================================
// __exit - функция для выгрузки

static void __exit var1_exit(void)
{
    // Составляем номер устройства из major и минора (0)
    dev_t dev = MKDEV(major, 0);

    // Удаляем устройство из /dev
    device_destroy(var1_class, dev);
    
    // Уничтожаем класс
    class_destroy(var1_class);
    
    // Удаляем символьное устройство из ядра
    cdev_del(&var1_cdev);
    
    // Освобождаем зарегистрированный номер устройства
    unregister_chrdev_region(dev, 1);

    // Освобождаем память, выделенную под массив results
    // kfree(NULL) допустимо - ничего не делает
    kfree(results);
}

// ============================================================
// БЛОК 10: РЕГИСТРАЦИЯ ТОЧЕК ВХОДА МОДУЛЯ
// ============================================================
// Указываем ядру, какая функция вызывается при загрузке (insmod)
module_init(var1_init);

// Указываем, какая функция вызывается при выгрузке (rmmod)
module_exit(var1_exit);

// ============================================================
// БЛОК 11: ЛИЦЕНЗИЯ И ИНФОРМАЦИЯ О МОДУЛЕ
// ============================================================
// Лицензия - обязательна, иначе ядро откажется загружать модуль
MODULE_LICENSE("GPL");

MODULE_AUTHOR("Student");
MODULE_DESCRIPTION("Variant 1 char device");
```

## Что делает эта программа (простыми словами)

1. **При загрузке** (`insmod var1.ko`) создаётся устройство `/dev/var1`

2. **Когда пользователь пишет** в это устройство (`write`):
   - Драйвер запоминает **длину** каждого вызова write
   - Например: записали 5 байт → в массиве появится число 5

3. **Когда пользователь читает** (`read`):
   - Драйвер отдаёт всё накопленные длины по одной на строку
   - Например: "100\n50\n30\n"

4. **При выгрузке** (`rmmod`) драйвер освобождает всю память

## Пример использования из пользовательской программы

```c
int fd = open("/dev/var1", O_RDWR);
write(fd, "hello", 5);   // драйвер запомнит 5
write(fd, "hi", 2);      // драйвер запомнит 2
close(fd);

fd = open("/dev/var1", O_RDONLY);
char buf[100];
int n = read(fd, buf, 100); 
// buf будет содержать "5\n2\n"
close(fd);
```
