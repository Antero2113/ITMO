## Файл `kernel.c`

```c
// Тип uint64 используется для хранения 64-битных значений.
// В OpenSBI большинство аргументов и результатов передаются именно так.
typedef unsigned long uint64;



// ===========================
// Работа с UART
// ===========================

// Базовый адрес UART-контроллера в QEMU virt machine.
// UART используется для вывода текста в консоль и чтения ввода пользователя.
#define UART_BASE 0x10000000

// UART_LSR (Line Status Register)
// Регистр состояния UART.
// Содержит флаги готовности передачи и приема данных.
#define UART_LSR  (*(volatile unsigned char *)(UART_BASE + 5))

// UART_THR (Transmit Holding Register)
// Регистр передачи данных.
// Запись байта в этот регистр отправляет символ в консоль.
#define UART_THR  (*(volatile unsigned char *)(UART_BASE + 0))



// =======================================================
// Функция вывода одного символа в UART
// =======================================================
void uart_putc(char c)
{
    // Проверяем бит 0x20 регистра UART_LSR.
    // Этот бит означает:
    // "передающий буфер свободен и готов принять новый символ".
    while (!(UART_LSR & 0x20));

    // Записываем символ в регистр передачи UART.
    UART_THR = c;
}



// =======================================================
// Функция получения одного символа из UART
// =======================================================
char uart_getc()
{
    // Проверяем бит 0x01 регистра UART_LSR.
    // Этот бит означает:
    // "получен новый символ".
    while (!(UART_LSR & 0x01));

    // Читаем символ из UART.
    return UART_THR;
}



// =======================================================
// Функция вывода строки
// =======================================================
// Посимвольно вызывает uart_putc()
// пока не встретится символ конца строки '\0'.
void print(const char *s)
{
    while (*s)
        uart_putc(*s++);
}



// =======================================================
// Вывод числа в шестнадцатеричном формате
// =======================================================
// Используется для отображения значений,
// возвращаемых функциями OpenSBI.
void print_hex(uint64 v)
{
    char hex[] = "0123456789ABCDEF";

    print("0x");

    // Печатаем число по 4 бита (1 hex-символ)
    // начиная со старших битов.
    for (int i = 15; i >= 0; i--)
        uart_putc(hex[(v >> (i * 4)) & 0xF]);
}



// ===========================
// Константы OpenSBI
// ===========================

// BASE extension
#define SBI_EXT_BASE 0x10

// Функция получения версии реализации SBI
#define SBI_BASE_GET_IMP_VERSION 0x2



// HSM extension (Hart State Management)
#define SBI_EXT_HSM 0x48534D

// Получение состояния hart
#define SBI_HART_GET_STATUS 0x0

// Остановка hart
#define SBI_HART_STOP 0x1



// System Reset extension
#define SBI_EXT_SRST 0x53525354

// Системное выключение/перезагрузка
#define SBI_SYSTEM_RESET 0x0



// =======================================================
// Универсальная функция вызова OpenSBI
// =======================================================
//
// Все вызовы OpenSBI выполняются через инструкцию ECALL.
//
// Используемые регистры:
//
// a0-a5 — аргументы функции
// a6    — Function ID (FID)
// a7    — Extension ID (EID)
//
// После вызова:
// a0 — код ошибки / результат
// a1 — дополнительное возвращаемое значение
//
static inline uint64 sbi_call(uint64 eid, uint64 fid,
                              uint64 arg0, uint64 arg1,
                              uint64 arg2, uint64 arg3,
                              uint64 arg4, uint64 arg5)
{
    // Аргументы SBI
    register uint64 a0 asm("a0") = arg0;
    register uint64 a1 asm("a1") = arg1;
    register uint64 a2 asm("a2") = arg2;
    register uint64 a3 asm("a3") = arg3;
    register uint64 a4 asm("a4") = arg4;
    register uint64 a5 asm("a5") = arg5;

    // Идентификатор функции SBI
    register uint64 a6 asm("a6") = fid;

    // Идентификатор расширения SBI
    register uint64 a7 asm("a7") = eid;

    // ECALL — переход в OpenSBI
    asm volatile("ecall"
                 : "+r"(a0), "+r"(a1)
                 : "r"(a2), "r"(a3), "r"(a4), "r"(a5),
                   "r"(a6), "r"(a7)
                 : "memory");

    // Возвращаем результат из регистра a0
    return a0;
}



// =======================================================
// Получение версии реализации SBI
// =======================================================
void get_version()
{
    uint64 r = sbi_call(
                        SBI_EXT_BASE,
                        SBI_BASE_GET_IMP_VERSION,
                        0,0,0,0,0,0
                        );

    print("\nSBI Implementation Version: ");
    print_hex(r);
    print("\n");
}



// =======================================================
// Получение состояния hart
// =======================================================
//
// Hart — аппаратное ядро процессора RISC-V.
//
void hart_get_status()
{
    print("\nEnter hart id (0-9): ");

    // Читаем номер hart с клавиатуры
    char c = uart_getc();

    // Отображаем введенный символ обратно в консоль
    uart_putc(c);

    // Перевод ASCII-символа в число
    int hart = c - '0';

    // Вызов OpenSBI
    uint64 r = sbi_call(
                        SBI_EXT_HSM,
                        SBI_HART_GET_STATUS,
                        hart,0,0,0,0,0
                        );

    print("\nHart status: ");
    print_hex(r);
    print("\n");
}



// =======================================================
// Остановка текущего hart
// =======================================================
void hart_stop()
{
    print("\nStopping hart...\n");

    sbi_call(
             SBI_EXT_HSM,
             SBI_HART_STOP,
             0,0,0,0,0,0
             );

    print("Hart stopped.\n");
}



// =======================================================
// Выключение виртуальной машины
// =======================================================
void system_shutdown()
{
    print("\nShutdown...\n");

    sbi_call(
             SBI_EXT_SRST,
             SBI_SYSTEM_RESET,
             0,0,0,0,0,0
             );
}



// =======================================================
// Главная функция программы
// =======================================================
//
// После загрузки ядра управление передается сюда.
// Программа работает бесконечно,
// отображая пользовательское меню.
//
void main()
{
    while (1)
    {
        print("\n");

        // Главное меню
        print("1. Get SBI implementation version\n");
        print("2. Hart get status\n");
        print("3. Hart stop\n");
        print("4. System Shutdown\n");

        print("Choose: ");

        // Читаем выбор пользователя
        char c = uart_getc();

        // Показываем введенный символ
        uart_putc(c);

        print("\n");

        // Вызываем нужную функцию
        if (c == '1')
            get_version();

        else if (c == '2')
            hart_get_status();

        else if (c == '3')
            hart_stop();

        else if (c == '4')
            system_shutdown();

        else
            print("Invalid option!\n");
    }
}
```

## Файл `start.S`

```asm
// Секция с кодом загрузки ядра
.section .text.boot

// Делаем символ boot видимым для линковщика
.globl boot



// =======================================================
// Точка входа в ядро
// =======================================================
//
// После запуска OpenSBI передает управление сюда.
//
boot:

    // Загружаем адрес вершины стека в регистр sp
    //
    // sp (stack pointer) — указатель стека.
    // Стек используется для хранения:
    // - локальных переменных
    // - адресов возврата функций
    // - временных данных
    la sp, stack_top

    // Переход в main()
    call main



// =======================================================
// Бесконечный цикл
// =======================================================
//
// Если main завершится,
// процессор останется здесь.
//
1:
    j 1b
```

## Файл `kernel.ld`

```ld
// =======================================================
// Linker script
// =======================================================
//
// Определяет:
// - адрес загрузки ядра
// - расположение секций
// - точку входа программы
//

// Точка входа в программу
ENTRY(boot)

SECTIONS
{
    // Адрес загрузки ядра в памяти
    . = 0x80200000;

    // ===================================================
    // Секция кода программы
    // ===================================================
    .text : {

        // Код загрузки размещается первым
        *(.text.boot)

        // Остальной код
        *(.text*)
    }

    // Секция констант
    .rodata : {
        *(.rodata*)
    }

    // Инициализированные глобальные переменные
    .data : {
        *(.data*)
    }

    // Неинициализированные глобальные переменные
    .bss : {
        *(.bss*)
        *(COMMON)
    }

    // Выравнивание адреса
    . = ALIGN(8);

    // ===================================================
    // Стек ядра
    // ===================================================
    //
    // stack_top — символ,
    // используемый в start.S
    //
    // Размер стека: 128 КБ
    stack_top = . + 128K;
}
```

## Файл `Makefile`

```makefile
# =======================================================
# Компилятор RISC-V
# =======================================================
CC=riscv64-unknown-elf-gcc



# =======================================================
# Параметры компиляции
# =======================================================

# -march=rv32imac
#   Архитектура RV32IMAC:
#   I  - базовые инструкции
#   M  - умножение/деление
#   A  - атомарные операции
#   C  - compressed инструкции
#
# -mabi=ilp32
#   ABI для 32-битной системы
#
# -ffreestanding
#   Сборка без стандартной библиотеки
#
# -nostdlib
#   Не подключать libc
#
# -O0
#   Отключение оптимизаций
CFLAGS=-march=rv32imac -mabi=ilp32 -ffreestanding -nostdlib -O0



# Linker script
LDFLAGS=-T kernel.ld



# =======================================================
# Сборка проекта
# =======================================================
all: kernel.elf



# Сборка ELF-файла ядра
kernel.elf: start.S kernel.c
	$(CC) $(CFLAGS) start.S kernel.c $(LDFLAGS) -o kernel.elf



# Очистка проекта
clean:
	rm -f *.elf
```

## Файл `run.sh`

```bash
#!/bin/bash

# =======================================================
# Запуск виртуальной машины QEMU
# =======================================================

qemu-system-riscv32 \

  # Виртуальная RISC-V платформа
  -machine virt \

  # Работа без графического окна
  # Ввод/вывод идет через терминал
  -nographic \

  # Использование встроенного OpenSBI
  -bios default \

  # Загружаемое ядро
  -kernel kernel.elf \

  # Подключение UART к текущему терминалу
  -serial mon:stdio

# При необходимости можно включить несколько ядер:
# -smp 2
```
